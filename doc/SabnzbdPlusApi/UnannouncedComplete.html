def process
  return_array = []
  history = @mapper.history
  
  history.slots.each { |slot|
    unless @announced_nzos.include? slot.nzo_id
      return_array = return_array << slot
      @announced_nzos = @announced_nzos << slot.nzo_id
    end
  }

  return return_array
end
