def self.from_hash(slot)
  item = super(slot)

  if slot.key? "stage_log"
    item.stage_log = []

    slot["stage_log"].each { |log|
      item.stage_log = item.stage_log << StageLog.from_hash(log)
    }
  end

  return item
end
