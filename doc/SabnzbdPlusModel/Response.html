def self.from_hash(response)
  item = self.new

  response.each { |key, value|
    unless self.parameter_mapping.key? key
      item.send(key.to_s+"=", value)
    else
      item.send(self.parameter_mapping[key].to_s+"=", value)
    end

  }

  return item
end
